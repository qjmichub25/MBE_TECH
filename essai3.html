async function createNewRecord(tableName) { if(!AIRTABLE_TOKEN) return alert("Veuillez renseigner votre Jeton d'accès."); const internalName = Object.keys(AT_TABLES).find(key => AT_TABLES[key] === tableName).toLowerCase(); // On prépare un nom par défaut selon la table pour éviter les lignes complètement vides let defaultFields = {}; if (tableName === "Qui") { defaultFields = { "Prenom Nom": "Nouveau Profil" }; } else if (tableName === "Statu" || tableName === "Sacrements" || tableName === "Evenement") { defaultFields = { "Name": "Nouvelle entrée" }; } try { const response = await fetch(`https://api.airtable.com/v0/${AIRTABLE_BASE_ID}/${encodeURIComponent(tableName)}`, { method: "POST", headers: { "Authorization": `Bearer ${AIRTABLE_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify({ fields: defaultFields }) // On envoie les données par défaut }); if(!response.ok) throw new Error(`Code HTTP ${response.status}`); const newRecord = await response.json(); if(tables[internalName]) { // On ajoute la ligne à l'écran avec les mêmes données par défaut tables[internalName].addRow({ airtable_id: newRecord.id, ...defaultFields }, true); // Le "true" force la ligne à s'ajouter TOUT EN HAUT du tableau pour la voir tout de suite } } catch (err) { alert(`Erreur lors de la création de la ligne : ${err.message}`); } }
